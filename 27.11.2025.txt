 package collection;

import java.util.*;

public class collectionarray 
 {
    public static void main(String[] args) {

        ArrayList<Integer> arrayList = new ArrayList<>();
        for (int i = 1; i <= 10; i++) {
            arrayList.add(i);
        }
    
        LinkedList<Integer> linkedList = new LinkedList<>();
        for (int i = 11; i <= 20; i++) {
            linkedList.add(i);
        }
        System.out.println("ArrayList Elements:");
        long startArrayTime = System.currentTimeMillis();

        Iterator<Integer> arrayIterator = arrayList.iterator();
        while (arrayIterator.hasNext()) {
            System.out.print(arrayIterator.next() + " ");
        }

        long endArrayTime = System.currentTimeMillis();
        long arrayTime = endArrayTime - startArrayTime;
        System.out.println("\nArrayList Iteration Time: " + arrayTime + " ms");

        System.out.println("\nLinkedList Elements:");
        long startLinkedTime = System.currentTimeMillis();

        Iterator<Integer> linkedIterator = linkedList.iterator();
        while (linkedIterator.hasNext()) {
            System.out.print(linkedIterator.next() + " ");
        }

        long endLinkedTime = System.currentTimeMillis();
        long linkedTime = endLinkedTime - startLinkedTime;
        System.out.println("\nLinkedList Iteration Time: " + linkedTime + " ms");
    }
}
add array,first,last,addall:
package addArray;
import java.util.*;

public class Array
{
    public static void main(String[] args) {

        LinkedList<Integer> list = new LinkedList<>();
        list.add(10);
        list.add(20);
        list.add(30);
        list.add(40);

        System.out.println("Original LinkedList: " + list);

        list.addFirst(5);
        System.out.println("After addFirst(): " + list);

        list.addLast(50);
        System.out.println("After addLast(): " + list);

        List<Integer> extraValues = Arrays.asList(60, 70, 80);
        list.addAll(extraValues);
        System.out.println("After addAll(): " + list);

        System.out.println("\nPrinting using Iterator:");
        Iterator<Integer> it = list.iterator();
        while (it.hasNext()) {
            System.out.print(it.next() + " ");
        }
    }
}

        list.add(30);

        list.add(40);

        System.out.println("Original LinkedList: " + list);

        list.removeFirstOccurrence(20);
        System.out.println("After removeFirstOccurrence(20): " + list);

        list.removeLastOccurrence(20);
        System.out.println("After removeLastOccurrence(20): " + list);

        list.removeAll(Arrays.asList(20));
        System.out.println("After removeAll(20): " + list);

        System.out.println("\nPrinting using Iterator:");
